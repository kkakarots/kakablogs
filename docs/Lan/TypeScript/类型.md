# void

```ts
  /** 
   * 这里的void表示忽略返回值
   * setSelectedLanguage()返回一个Promise, 不加void会ts报红
   * 防止 ESLint 或其他工具报告"未使用的返回值"警告
   */
  const onChangeLanguage = useCallback(
    (event: SelectChangeEvent<Language>) => {
      const lang = event.target.value as Language;
      void setSelectedLanguage(lang); // 忽略返回值
      i18n.changeLanguage(lang).catch((error: unknown) => {
        console.error("Failed to switch languages", error);
        reportError(error as Error);
      });
    },
    [i18n, setSelectedLanguage],
  );
```

# 私有属性(方法)
``` TypeScript
class MyClass {
  // 1. ES 原生私有方法（# 开头）
  #nativePrivateMethod() {
    console.log("ES 原生私有方法");
  }

  // 2. TypeScript 语法糖私有方法（private 关键字）
  private tsPrivateMethod() {
    console.log("TS 编译时私有方法");
  }

  public callPrivateMethods() {
    this.#nativePrivateMethod(); // 类内部可访问
    this.tsPrivateMethod(); // 类内部可访问
  }
}

const instance = new MyClass();
instance.callPrivateMethods(); // 正常执行

// 外部访问测试
instance.#nativePrivateMethod(); // ❌ 报错：Property '#nativePrivateMethod' is not accessible outside class 'MyClass' because it has a private identifier
instance.tsPrivateMethod(); // ❌ 编译时报错：Property 'tsPrivateMethod' is private and only accessible within class 'MyClass'

// 运行时绕过 TS 的 private 限制（# 方法无法绕过）
(instance as any).tsPrivateMethod(); // ✅ 运行时可执行（TS 语法糖的漏洞）
(instance as any).#nativePrivateMethod(); // ❌ 运行时仍报错（真正私有）
```
